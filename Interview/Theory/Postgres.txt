GROUP BY + HAVING
--
SELECT department, COUNT(*) 
FROM employees
WHERE salary > 50000
GROUP BY department
HAVING COUNT(*) > 5
ORDER BY COUNT(*) DESC;
--
order of pheonix
Filter → Group → Aggregate → Filter groups (HAVING) → Sort → Done.
