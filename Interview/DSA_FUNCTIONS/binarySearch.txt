
 public static int higherbound(List<Integer> list, int target)
    {
        int low = 0;
        int high = list.size();
        while (low < high)
        {
            int mid = low + (high - low) / 2;
            if (list.get(mid) <= target)
            {
                low = mid + 1;
            }
            else
            {
                high = mid;
            }
        }
        return low;
    }
